관계형데이터가 아닌 NoSQL

NoSQL: 키-값 DB (Redis)

  • 형식: 단순히 Key와 Value 쌍으로 저장
  • 대표 제품: Redis (메모리 기반, 매우 빠른 성능)
  • 활용: 세션 관리, 캐시, 실시간 랭킹
# Redis 명령어 예시
SET user:1001 "{'name':'홍길동','age':20,'major':'컴퓨터공학'}"
SET user:1002 "{'name':'김철수','age':22,'major':'경영학'}"

GET user:1001
-- 결과: {'name':'홍길동','age':20,'major':'컴퓨터공학'}

NoSQL: 문서(Document) DB

  • 형식: JSON 문서 형태로 데이터 저장
  • 대표 제품: MongoDB, Oracle 23ai(JSON 컬럼 지원)
  • 활용: 사용자 프로필, 로그 데이터, 웹/앱 데이터 저장
-- Oracle 23ai JSON 예시
CREATE TABLE students (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY,
  info JSON
);

INSERT INTO students (info) VALUES ('{
  "student_id": 1001,
  "name": "홍길동",
  "age": 20,
  "major": "컴퓨터공학"
}');

SELECT info.name, info.age FROM students;

NoSQL: 그래프(Graph) DB

  • 형식: 노드(Node)와 엣지(Edge)로 관계 표현
  • 대표 제품: Neo4j, Oracle 23ai Graph
  • 활용: SNS 친구 관계, 추천 시스템, 네트워크 분석
-- Oracle 23ai Graph 예시 (PGQL 쿼리)
INSERT VERTEX student1 LABELS (Student) PROPERTIES (name='홍길동');
INSERT VERTEX student2 LABELS (Student) PROPERTIES (name='김철수');
INSERT VERTEX course1  LABELS (Course)  PROPERTIES (title='데이터베이스');

INSERT EDGE enrolled BETWEEN student1 AND course1 LABELS (수강);
INSERT EDGE friend   BETWEEN student1 AND student2 LABELS (친구);

-- 관계 조회
SELECT s.name, c.title
FROM MATCH (s:Student)-[수강]->(c:Course);

Oracle 23ai: 벡터(Vector) 기반 저장

  • 형식: 데이터를 임베딩(Embedding) 벡터로 변환해 저장
  • 특징: 유사도 검색(Similarity Search) 지원 → 의미적으로 가까운 데이터 검색 가능
  • 활용: 추천 시스템, 챗봇 검색, 이미지/문서 유사도 탐색
-- 벡터 컬럼을 포함한 테이블 생성
CREATE TABLE documents (
  id NUMBER GENERATED BY DEFAULT AS IDENTITY,
  content VARCHAR2(200),
  embedding VECTOR(768)  -- 768차원 벡터
);

-- 데이터 삽입
INSERT INTO documents (content, embedding)
VALUES ('AI 개요 문서', VECTOR('[0.12, 0.98, -0.33, ...]'));

INSERT INTO documents (content, embedding)
VALUES ('데이터베이스 수업 자료', VECTOR('[0.10, 0.95, -0.30, ...]'));

-- 쿼리 벡터와 유사한 문서 검색
SELECT content
FROM documents
ORDER BY embedding <-> VECTOR('[0.11, 0.96, -0.31, ...]')
FETCH FIRST 3 ROWS ONLY;
